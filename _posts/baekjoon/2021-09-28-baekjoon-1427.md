## 3. 풀이

```cpp
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int compare(int i, int j){
    if(i>j) return true;
    else return false;
}

int main(){
    int number; cin>>number;
    vector<int> v;
    
    while (number != 0)
    {
        v.push_back(number%10);
        number /= 10;
    }

    // 내림차순 정렬 
    sort(v.begin(), v.end(), compare);
    
    for(int i=0;i<v.size();i++) cout<<v[i];

}
```
